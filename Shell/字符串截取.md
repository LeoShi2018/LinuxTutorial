# 字符串截取

### 1. 字符串左删除#
nodeip02=10.1.0.142
<pre>
# 从左到右删除*到第一个分隔符.
# echo ${nodeip02#*.}
----------
    1.0.142

# 从左到右删除*到最后一个分隔符
# echo ${nodeip02##*.}
----------
    142
</pre>

### 2. 字符串右删除%
<pre>
# 从右到左删除*到第一个分隔符.
# echo ${nodeip02%.*} 
----------
    10.1.0
# 从右到左删除*到最后一个分隔符
# echo ${nodeip02%%.*}
----------
    10
</pre>

### 3. 指定截取
<pre>
# 跟数组切片类似0代表第1个字符，3代表从左取3个
# echo ${nodeip02:0:3}
----------
    10.
# echo ${nodeip02:2:5}
----------
    .1.0.
# 从左边数第6个字符(数5不含5)开始右取到头
# echo ${nodeip02:5}
----------   
    0.142
# 从右往左数到第3个字符，再往右截取2个字符
# echo ${nodeip02:0-3:2}
----------
    14
# aaa=www.baidu.com
# echo ${aaa:0-9:5}
----------     
    baidu
# echo ${aaa:0-3:3}
----------
    com
</pre>
